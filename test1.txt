#include <stdio.h>   // Подключение стандартной библиотеки ввода-вывода (для printf, scanf)
#include <math.h>    // Подключение математической библиотеки (для sqrt)

int main() {
    // Объявление переменных:
    double a, b, c;          // Коэффициенты квадратного уравнения ax² + bx + c = 0
    double discriminant;     // Дискриминант уравнения
    double root1, root2;     // Корни уравнения (если существуют)

    // Ввод коэффициентов:
    printf("Введите коэффициенты a, b и c: ");  // Подсказка для пользователя
    scanf("%lf %lf %lf", &a, &b, &c);           // Чтение трёх вещественных чисел

    // Вычисление дискриминанта (D = b² - 4ac):
    discriminant = b * b - 4 * a * c;

    // Проверка значения дискриминанта:
    if (discriminant > 0) {  
        // Случай 1: D > 0 → два различных действительных корня
        root1 = (-b + sqrt(discriminant)) / (2 * a);  // Первый корень
        root2 = (-b - sqrt(discriminant)) / (2 * a);  // Второй корень
        printf("Два различных корня: %.2lf и %.2lf", root1, root2);  // Вывод с округлением до 2 знаков
    }
    else if (discriminant == 0) {  
        // Случай 2: D = 0 → один корень (кратный)
        root1 = -b / (2 * a);  // Корень единственный
        printf("Один корень: %.2lf", root1);
    }
    else {  
        // Случай 3: D < 0 → действительных корней нет
        printf("Действительных корней нет");
    }

    return 0;  // Завершение программы с кодом 0 (успех)
}
